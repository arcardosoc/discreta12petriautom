230 void gerar_imagem(transicao *tr, int ql, int qt) /** Função utilizada para desenhar a rede de petri utilizando a biblioteca do allegro. */
231 {
232  BITMAP *buff;
233  PALETTE pal;
234  arco *pt = tr->entram; /** Variavel temporaria para não modificar a posição da cabeça. */
235  arco *pl = tr->saem; /** Variavel temporaria para não modificar a posição da cabeça. */
236  int k=1,flag;
237 
238  if(install_allegro(SYSTEM_NONE, &errno, atexit) !=0) /** Inicialização do allegro e com verificação de falhas. */
239  exit(EXIT_FAILURE);
240  set_color_depth(16); /** Setagem no numero de cores. */
241  get_palette(pal);
242 
243  buff = create_bitmap(X,Y); /** Criação da imagem de tamanho X e Y, que foram previamentes definidos. */
244  if(buff == NULL) /** Verificação de falhas na criação da imagem. */
245  {
246  printf("Nao foi possivel criar a imagem!\n");
247  exit(EXIT_FAILURE);
248  }
249  desenha_estados(buff,ql); /** Chamada da função desenha_estados. */
250  desenha_transicoes(buff, tr, ql, qt); /** Chamada da função desenha_transições. */
251 
252 
253  while(1) /** Laço utilizado para transitar na lista de transições. */
254  {
255  flag=1; /** Flag que será usada como comparativo na função desenha_arcos. */
256  while(pt !=NULL) /** Laço utilizado para transitar na lista de arco entram. */
257  {
258  desenha_arcos(pt->inicio*2,pt->final+k,buff,ql*2,pt->tkgp,flag); /** Chamada da função desenha arcos voltada para arcos lugar-transição. */
259  pt = pt->prox; /** Método utilizado para transitar pela lista arco entram. */
260  }
261  flag=0; /** Flag que será usada como comparativo na função desenha_arcos. */
262  while(pl !=NULL ) /** Laço utilizado para transitar na lista de arco saem. */
263  {
264  desenha_arcos(pl->inicio+k,pl->final*2,buff,ql*2,pl->tkgp,flag); /** Chamada da função desenha arcos voltada para arcos transição-lugar. */
265  pl = pl->prox; /** Método utilizado para transitar pela lista arco saem. */
266  }
267  k++; /** Variavel utilizada para que osvalores das transições sempre entrem como impares nas funções. */
268  tr = tr->prox; /** Método utilizado para transitar pela lista transição. */
269  if(tr == NULL) /** Codição de verificação para ver se a lista transição acabou. */
270  break;
271  pt = tr->entram; /** Voltar o ponteiro para o inicio da lista arco entram da nova transição. */
272  pl = tr->saem; /** Voltar o ponteiro para o inicio da lista arco saem da nova transição. */
273 
274  }
275  save_bitmap(IMAGENAME, buff, pal); /** Salvar a imagem desenhada. */
276  destroy_bitmap(buff);
277  allegro_exit(); /** Terminar o allegro. */
278 
279  printf("Imagem %s salva com sucesso!\n", IMAGENAME);
280 }

317 void retiratoken(lugartoken **cabeca, arco *head, arco *kopf)
318 {
319  lugartoken *pl=*cabeca; /** Variavel temporaria para não modificar a posição da cabeça. */
320  arco *pt=head; /** Variavel temporaria para não modificar a posição da cabeça. */
321  while(pt != NULL) /** Laço utilizado para transitar na lista de arco saem. */
322  {
323  while(pl != NULL) /** Laço utilizado para transitar na lista de arco saem. */
324  {
325  if(pt->inicio == pl->lu && pl->tk >= pt->tkgp) /** Condição de igualdade entre lugar incial do arco lugar-transição e lugar da lista de tokens e verificação para saber se o numero de tokens no lugar é maior ou igual ao numero de tokens pedidos pelo arco lugar-transição. */
326  {
327  if(DEBUG > 4 )
328  printf("Pthread[%d]:\nHouve retirada de token\n\n",pt->final);
329  printf("pt->inicio:%d = pl->lu:%d\npl->tk:%d - pt->tkgp:%d\n",pt->inicio,pl->lu,pl->tk,pt->tkgp);
330  pl->tk-=pt->tkgp; /** Subtração do numero de tokens de acordo com o numero de tokens pedidos pelo arco lugar-transição. */
331  if(rand()%100+1 < PAT) /** Condição de ativação da transição, 50% de chance de ativação. */
332  {
333  if(DEBUG > 4)
334  printf("Pthread[%d]:\nTransicao Ativada com Sucesso\n\n",pt->final);
335  ativacaotransicao(kopf,&lntk); /** Chamada da função ativacaotransicao. */
336  }
337  }
338  if(DEBUG > 4 && pt->inicio == pl->lu && pl->tk > pt->tkgp)
339  printf("Pthread[%d]:\nNao houve retirada de token\n\n",pt->final);
340  pl = pl->prox; /** Método utilizado para transitar pela lista lugartoken. */
341  }
342  pl=*cabeca; /** Voltar o ponteiro para o inicio da lista lugartoken. */

436 {

438  float xi,yi,rc,raio;
439  transicao *pl=trans;
440  raio = (Y/8)*(M_PI/(M_PI+k));
441  rc = YCentro - raio*4;
442  raio = (Y/12)*(M_PI/(M_PI+k));
443 
444  while(1) /** Laço utilizado para transitar pela lista de transiçoes. */
445  {
446  for(i=j;i<c*2;i++) /** Laço utilizado para desenhar as transições. */

573  return M_PI/2.0;

575  }
